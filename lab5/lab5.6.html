Завдання:

1.Створити одновимірний масив, кількість елементів якого задана користувачем. Знайти суму елементів з парними індексами, максимальний елемент та його індекс, мінімальний елемент та його індекс серед елементів з непарними індексами.

2. Упорядкувати масив у порядку зростання, Надрукувати вхідний та вихідний масив.

Застосувати функції

Виконання коду

Код:

            // Функція для створення масиву за кількістю елементів, введених користувачем
            function createArray(length) {
                var arr = [];
                for (var i = 0; i < length; i++) {
                    arr.push(Math.floor(Math.random() * 100)); // Генеруємо випадкові елементи в межах від 0 до 99
                }
                return arr;
            }
            
            // Функція для знаходження суми елементів з парними індексами
            function sumOfEvenIndexElements(arr) {
                var sum = 0;
                for (var i = 0; i < arr.length; i += 2) { // Починаємо з 0 та кроком 2
                    sum += arr[i];
                }
                return sum;
            }
            
            // Функція для знаходження максимального та мінімального елементів з непарними індексами
            function maxMinOfOddIndexElements(arr) {
                var max = -Infinity;
                var min = Infinity;
                var maxIndex, minIndex;
            
                for (var i = 1; i < arr.length; i += 2) { // Починаємо з 1 та кроком 2
                    if (arr[i] > max) {
                        max = arr[i];
                        maxIndex = i;
                    }
                    if (arr[i] < min) {
                        min = arr[i];
                        minIndex = i;
                    }
                }
            
                return { max: max, maxIndex: maxIndex, min: min, minIndex: minIndex };
            }
            
            // Функція для сортування масиву у порядку зростання
            function sortArray(arr) {
                return arr.slice().sort(function(a, b) {
                    return a - b;
                });
            }
            
            // Вхідний масив
            var length = parseInt(prompt("Введіть кількість елементів масиву:"));
            var array = createArray(length);
            console.log("Вхідний масив:", array);
            
            // Виконання першого завдання
            var sumOfEven = sumOfEvenIndexElements(array);
            var { max: maxOdd, maxIndex: maxOddIndex, min: minOdd, minIndex: minOddIndex } = maxMinOfOddIndexElements(array);
            console.log("Сума елементів з парними індексами:", sumOfEven);
            console.log("Максимальний елемент з непарними індексами:", maxOdd, "і його індекс:", maxOddIndex);
            console.log("Мінімальний елемент з непарними індексами:", minOdd, "і його індекс:", minOddIndex);
            
            // Виконання другого завдання
            var sortedArray = sortArray(array);
            console.log("Відсортований масив у порядку зростання:", sortedArray);
